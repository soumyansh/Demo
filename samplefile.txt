import java.nio.charset.StandardCharsets;

public class TextToHex {
    public static void main(String[] args) {
        String text = "Hello, World!";
        String hex = stringToHex(text);
        System.out.println("Text: " + text);
        System.out.println("Hex: " + hex);
    }

    public static String stringToHex(String text) {
        byte[] bytes = text.getBytes(StandardCharsets.UTF_8);
        StringBuilder hexString = new StringBuilder();

        for (byte b : bytes) {
            // Convert byte to unsigned integer before formatting
            int unsignedByte = b & 0xff;
            // Convert unsigned integer to hexadecimal
            String hex = Integer.toHexString(unsignedByte);
            // Ensure the hex representation is two characters long
            if (hex.length() == 1) {
                hex = "0" + hex;
            }
            hexString.append(hex);
        }

        return hexString.toString().toUpperCase(); // Convert to uppercase for consistency
    }
}
